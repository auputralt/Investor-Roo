<!-- Application Process -->
<section class="section-sm">
  <div class="container">
    <h2>Application Process</h2>
    
    <div class="timeline">
      <div class="timeline-item">
        <div class="timeline-date">Step 1</div>
        <div class="timeline-title">Farmer Group Formalization</div>
        <div class="timeline-content">
          <ul>
            <li>Ensure kelompok tani have NIK, NPWP, land documentation</li>
            <li>Register in SIMLUHTAN system</li>
            <li>Prepare RDKK (Rencana Definitif Kebutuhan Kelompok)</li>
          </ul>
        </div>
      </div>
      
      <div class="timeline-item">
        <div class="timeline-date">Step 2</div>
        <div class="timeline-title">Dinas Pertanian Coordination</div>
        <div class="timeline-content">
          <ul>
            <li>Submit proposals to Dinas Pertanian Kabupaten</li>
            <li>Dinas aggregates and prioritizes proposals</li>
            <li>Review against annual Juknis requirements</li>
          </ul>
        </div>
      </div>
      
      <div class="timeline-item">
        <div class="timeline-date">Step 3</div>
        <div class="timeline-title">Province-Level Review</h3>
        <div class="timeline-content">
          <ul>
            <li>Dinas Provinsi reviews and ranks proposals</li>
            <li>Alignment with provincial priorities</li>
            <li>Coordination with Bappeda for budget allocation</li>
          </ul>
        </div>
      </div>
      
      <div class="timeline-item">
        <div class="timeline-date">Step 4</div>
        <div class="timeline-title">Kementan Submission</div>
        <div class="timeline-content">
          <ul>
            <li>Proposals submitted to relevant DG (Tanaman Pangan, Perkebunan, PSP)</li>
            <li>National-level review and approval</li>
            <li>Budget allocation through APBN process</li>
          </ul>
        </div>
      </div>
    </div>
  </div>
</section>
